cmake_minimum_required(VERSION 3.30.3)
project(freeglutRotate)

#使用c++11来编译代码
set(CMAKE_CXX_STANDARD 11)

#包含头文件目录
include_directories(${CMAKE_SOURCE_DIR}/Hfile)
include_directories(${CMAKE_SOURCE_DIR}/freeglut/include)
include_directories(${CMAKE_SOURCE_DIR}/glew/include)
#设置源文件
set(SOURCE_FILES 
src/main.cpp)  

#添加可执行文件
add_executable(rotate ${SOURCE_FILES})
